This is the answer of Exercise #9: No for-loops.

 

Question #6: Widget::DoIt on Widget*

 

Replace the for-loop. You will need:

  • std::for_each

  • std::mem_fun (or boost::mem_fn)

     

     

     

    #include <vector>

     

    struct Widget

    {

    void DoIt() const { /* do it */ }

    };

     

    void DoIt(const std::vector<Widget*>& v)

    {

    const int sz = v.size();

    for (int i=0; i!=sz; ++i)

    {

    v[i]->DoIt();

    }

    }

     

     

    Answer STL-only

     

     

    #include <vector>

    #include <algorithm>

    #include <numeric>

     

    struct Widget

    {

    void DoIt() const { /* do it */ }

    };

     

    void DoIt(const std::vector<Widget*>& v)

    {

    std::for_each(v.begin(),v.end(),std::mem_fun(&Widget::DoIt));

    }

     

     

    Answer using Boost

     

     

    #include <vector>

    #include <algorithm>

    #include <boost/mem_fn.hpp>

     

    struct Widget

    {

    void DoIt() const { /* do it */ }

    };

     

    void DoIt(const std::vector<Widget*>& v)

    {

    std::for_each(v.begin(),v.end(),boost::mem_fn(&Widget::DoIt));

    }

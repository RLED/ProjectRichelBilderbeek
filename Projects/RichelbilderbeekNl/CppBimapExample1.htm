main.cpp

 

#include <cassert>
#include <boost/bimap.hpp>

int main()
{
  boost::bimap<int,int> m;
  m.insert(boost::bimap<int,int>::value_type(1, 1));
  m.insert(boost::bimap<int,int>::value_type(2, 4));
  m.insert(boost::bimap<int,int>::value_type(3, 9));
  m.insert(boost::bimap<int,int>::value_type(4,16));
  m.insert(boost::bimap<int,int>::value_type(5,25));
  assert(m.left.find( 0) == m.left.end());
  assert(m.left.find( 5) != m.left.end());
  assert(m.left.find(16) == m.left.end());
  assert(m.right.find( 0) == m.right.end());
  assert(m.right.find( 5) == m.right.end());
  assert(m.right.find(16) != m.right.end());
}

#ifndef BILDERBIKKEL_QT_ASSERT_H
#define BILDERBIKKEL_QT_ASSERT_H

//From http://www.richelbilderbeek.nl/CppAssert.html
#ifdef NDEBUG_BILDERBIKKEL
  #define Assert(x) ((void)0)
#else
  #include <fstream>
  #include <iostream>
  #include <stdexcept>

  #define Assert(x)                      \
  if (!(x))                              \
  {                                      \
    std::cout                            \
      << "ERROR!! Assertion "            \
      <<  std::string (#x)               \
      <<  " failed\n on line "           \
      <<  (__LINE__)                     \
      <<  "\n in file "                  \
      <<  __.htm">FILE__                       \
      << std::endl;                      \
    std::ofstream f("assert_out.txt");   \
    f                                    \
      << "ERROR!! Assertion "            \
      <<  std::string (#x)               \
      <<  " failed\n on line "           \
      <<  (__LINE__)                     \
      <<  "\n in file "                  \
      <<  __.htm">FILE__                       \
      << std::endl;                      \
    f.close();                           \
      throw std::logic_error(            \
        "Assertion failed");             \
  }

#endif

#endif // BILDERBIKKEL_QT_ASSERT_H

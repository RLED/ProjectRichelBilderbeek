(C++) QGraphicsPixmapItem example 2: clickable pixmap

 

QGraphicsPixmapItem example 2: clickable pixmap is a QGraphicsPixmapItem example. This example shows how to put multiple movable QGraphicsPixmapItems on screen and let them respond to mouse clicks using a boost::signal.

qtitem.h

 

#ifndef QTITEM_H
#define QTITEM_H

#include <QGraphicsPixmapItem>
#include <boost/signals2.hpp>

///A QGraphicsPixmapItem that loads its pixmap from resources
///and is clickable
struct QtItem : public QGraphicsPixmapItem
{
  QtItem(QGraphicsItem *parent = 0, QGraphicsScene *scene = 0);

  ///Signal emitted when clicked
  boost::signals2::signal<void (QtItem*)> m_clicked_signal;

  protected:
  void mousePressEvent(QGraphicsSceneMouseEvent *event);
};

#endif // QTITEM_H

 

 

 

 

 

qtitem.cpp

 

#include <cassert>
#include <QPainter>
#include "qtitem.h"

QtItem::QtItem(QGraphicsItem *parent, QGraphicsScene *scene)
  : QGraphicsPixmapItem(parent,scene)
{
  assert(this->pixmap().isNull()
    && "Assume no pixmap loaded yet");

  //Load the pixmap from resources
  this->setPixmap(QPixmap(":/images/PicR.png"));

  assert(!this->pixmap().isNull()
    && "Assume pixmap is loaded successfully");

  this->setFlags(QGraphicsItem::ItemIsMovable);
}

void QtItem::mousePressEvent(QGraphicsSceneMouseEvent *)
{
  m_clicked_signal(this);
}

qtwidget.h

 

#ifndef QTWIDGET_H
#define QTWIDGET_H

#include <QGraphicsView>

///Forward declaration
struct QtItem;

struct QtWidget : public QGraphicsView
{
  QtWidget();

  ///Respond to a click on a QtItem
  void OnItemClick(QtItem * const item);
};

#endif // QTWIDGET_H

 

 

 

 

 

qtwidget.cpp

 

#include <cassert>

#include <boost/bind.hpp>
#include <boost/lambda/bind.hpp>

#include <QGraphicsScene>
#include <QGraphicsPixmapItem>

#include "qtitem.h"
#include "qtwidget.h"

QtWidget::QtWidget()
{
  QGraphicsScene * const scene = new QGraphicsScene(this);
  this->setScene(scene);

  const int n_items = 8;
  for (int i=0; i!=n_items; ++i)
  {
    QtItem * const item = new QtItem;
    item->m_clicked_signal.connect(
      boost::bind(
        &QtWidget::OnItemClick,this, boost::lambda::_1));
    //Scatter those items around a bit
    item->setPos(
      - 128 + (std::rand() % 256),
      - 128 + (std::rand() % 256));

    scene->addItem(item);
  }
}

void QtWidget::OnItemClick(QtItem * const item)
{
  item->setRotation(item->rotation() + 10.0);
}

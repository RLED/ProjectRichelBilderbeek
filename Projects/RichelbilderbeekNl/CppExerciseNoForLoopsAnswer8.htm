Question #8: GetSum

 

Replace the for-loop. You will need:

  • std::accumulate

     

     

     

    #include <vector>

     

    const int GetSum (const std::vector<int>& v)

    {

    const int sz = v.size();

    const int sum = 0;

    for (int i=0; i!=sz; ++i)

    {

    sum+=v[i];

    }

    return sum;

    }

    #include <vector>

    #include <algorithm>

    #include <numeric>

     

    const int GetSum (const std::vector<int>& v)

    {

    return std::accumulate(v.begin(),v.end(),0);

    }

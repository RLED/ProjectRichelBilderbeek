#include <iostream>
#include "ircbot.h"

const std::vector<std::string> Respond(const std::string& s);

int main()
{
  const std::string bot_name = "voidhosbn";
  const std::string server_name = "irc.zeronode.net";
  const std::string channel_name = "#Hive5";
  const int port = 6667;

  IrcBot(
    bot_name,
    channel_name,
    port,
    server_name,
    Respond
  );
}

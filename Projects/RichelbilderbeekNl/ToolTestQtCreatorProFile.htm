//See http://www.richelbilderbeek.nl/CppCompileErrorSwprintfHasNotBeenDeclared.htm
#undef __STRICT_ANSI__
#endif

#include <cassert>
#include <fstream>
#include <iostream>
#include <iterator>

#include <boost/filesystem.hpp>
#include <boost/program_options.hpp>

#include "qtcreatorprofile.h"
#include "testqtcreatorprofilemenudialog.h"
#include "trace.h"

int main(int argc, char* argv[])
{
  START_TRACE();

  boost::program_options::options_description d("Allowed options for ToolTestQtCreatorProFileConsole");
  std::string filename;

  d.add_options()
      ("about","displays the 'About' information")
      ("filename",
         boost::program_options::value<std::string>(&filename),
         "filename of Qt Creator project (.pro) file")
      ("help","produce this help message")
      ("version","displays the version")
       ;

  boost::program_options::variables_map m;
  boost::program_options::store(
    boost::program_options::parse_command_line(
      argc, argv, d), m);
  boost::program_options::notify(m);

  if (m.count("help"))
  {
    std::cout << d << "\n";
    return 0;
  }

  if (m.count("about"))
  {
    std::cout << TestQtCreatorProFileMenuDialog::GetAbout() << "\n";
    return 0;
  }

  if (m.count("version"))
  {
    std::cout << TestQtCreatorProFileMenuDialog::GetVersion() << "\n";
    return 0;
  }

  if (m.count("filename"))
  {
    if (filename.empty())
    {
      std::cout << "Please specify a filename.\n";
      return 1;
    }
    if (filename.size() < 5)
    {
      std::cout << "Please specify a full Qt Creator (.pro) project filename.\n";
      return 1;
    }
    if (filename.substr(filename.size() - 4,4) != ".pro")
    {
      std::cout << "Please specify a filename ending on .pro.\n";
      return 1;
    }
    if (!boost::filesystem::exists(filename))
    {
      std::cout << "Please specify a Qt Creator (.pro) project filename that exists.\n";
      return 1;
    }
    const boost::shared_ptr<const QtCreatorProFile> p(
      new QtCreatorProFile(filename));
    std::cout << p << '\n';
    return 0;
  }

  //Display the help
  std::cout << d << "\n";
}

ribi::QtTestQtCreatorProFileMainDialog::QtTestQtCreatorProFileMainDialog(QWidget *parent) :
  QtHideAndShowDialog(parent),
  ui(new Ui::QtTestQtCreatorProFileMainDialog)
{
  ui->setupUi(this);
}

ribi::QtTestQtCreatorProFileMainDialog::~QtTestQtCreatorProFileMainDialog() noexcept
{
  delete ui;
}

void ribi::QtTestQtCreatorProFileMainDialog::on_button_select_clicked()
{
  const std::string filename
    = QFileDialog::getOpenFileName(
      this,"Please select a .pro file",
      QString(),
      "Qt Project files (*.pro)"
      ).toStdString();

  if (!QFile::exists(filename.c_str()))
  {
    ui->text_result->clear();
    const std::string text = std::string("File '") + filename + std::string("' does not exist.");
    ui->text_result->setPlainText(text.c_str());
    return;
  }

  const boost::shared_ptr<const QtCreatorProFile> p(new QtCreatorProFile(filename));

  std::stringstream s;
  s << p;

  ui->text_result->clear();
  ui->text_result->setPlainText(s.str().c_str());
}

Question #17: Halve

 

Replace the for-loop. You will need:

  • std::bind2nd

  • std::divides

  • std::transform

     

     

    #include <vector>

     

    void Halve (std::vector<double>& v)

    {

    const int sz = v.size();

    for (int i=0; i!=sz; ++i)

    {

    v[i]/=2.0;

    }

    }

     

     

     

     

     

     

    Answer

     

     

    #include <vector>

    #include <algorithm>

    #include <numeric>

     

    //From http://www.richelbilderbeek.nl/CppHalve.htm

    void Halve (std::vector<double>& v)

    {

    std::transform(v.begin(),v.end(),v.begin(),

    std::bind2nd(std::divides<double>(),2.0));

    }
